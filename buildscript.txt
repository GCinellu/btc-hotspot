##
##  OS: 2016-09-23-raspbian-jessie-lite.img
##
##  Raspberry Pi3

ESSID="swagger"	# Hotspot ap name
IFACE_WAN=eth0  # Toward the internet
IFACE_LAN=wlan0 # Offered to local clients

# Update packages
apt-get update

# Get dependencies and tools
apt-get install -y \
  git \
  build-essential \
  libtool \
  autoconf \
  automake \
  gengetopt \
  devscripts \
  debhelper \
  libssl-dev \
  iptables \
  haserl \
  net-tools \
  libjson0 \
  libjson0-dev \
  freeradius \
  freeradius-mysql \
  hostapd \
  openjdk-8-jdk \
  mysql-server    


# Alter network interfaces  TODO: add WAN section
nano /etc/network/interfaces >>
"auto lo $IFACE_LAN
iface $IFACE_LAN inet static
address 10.0.0.1
netmast 255.255.255.0
gateway 10.0.0.1"

# Create hostapd conf
echo "auth_algs=1
channel=3
country_code=IT
disassoc_low_ack=1
# Require clients to know the network name
ignore_broadcast_ssid=0
hw_mode=g
interface=$IFACE_LAN
driver=nl80211
ieee80211n=1
ssid=$ESSID" > /etc/hostapd/hostapd.conf

# Edit hostapd init script to use the conf 
# BROKEN (must change only the first occurrence)!!!
sed -i '/DAEMON_CONF/c\DAEMON_CONF=/etc/hostapd/hostapd.conf/' /etc/init.d/hostapd

#sed -i '/Exec/c\Exec=spotify --force-device-scale-factor=1.5 %U' /usr/share/applications/spotify.desktop

# Allow ipv4 packet forwarding in kernel -- is that persistent?
echo 1 > /proc/sys/net/ipv4/ip_forward

# Enable hostapd at boot
systemctl enable hostapd
